function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); const data = JSON.parse(e.postData.contents); const userId = data.userId; const name = data.name; const rows = sheet.getDataRange().getValues(); let found = false; let points = 0; for (let i = 1; i < rows.length; i++) { // 👉 คอลัมน์ E = index 4 (LINE User ID) if (rows[i][4] === userId) { found = true; // 👉 คอลัมน์ G = index 6 (แต้ม) if (data.action === "add") { points = rows[i][6] + 1; sheet.getRange(i+1, 7).setValue(points); } else { points = rows[i][6]; } } } if (!found) { points = 1; // 👉 ใส่ข้อมูลลงตำแหน่งที่ถูก sheet.appendRow([ "", // Member ID "", // ลำดับ name, // ชื่อ (C) "", // เบอร์ userId, // LINE User ID (E) "", // ทริป points // แต้ม (G) ]); } return ContentService.createTextOutput(JSON.stringify({ points })) .setMimeType(ContentService.MimeType.JSON); }
